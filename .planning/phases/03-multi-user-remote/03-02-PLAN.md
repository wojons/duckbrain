  <action>Create Dockerfile for production deployment per D-05, D-07, D-20:

```dockerfile
# Build stage
FROM node:20-alpine AS builder
WORKDIR /app
COPY package*.json ./
RUN npm ci --only=production
COPY . .
RUN npm run build

# Production stage
FROM alpine:latest
RUN apk add --no-cache nodejs npm git

# Create non-root user (D-07, D-20)
RUN addgroup -g 1000 duckbrain && \
    adduser -u 1000 -G duckbrain -s /bin/sh -D duckbrain

WORKDIR /app

# Copy built application
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./

# Create data directory with proper ownership
RUN mkdir -p /data && chown -R duckbrain:duckbrain /data
VOLUME /data

# Switch to non-root user
USER duckbrain

EXPOSE 3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD node /app/dist/cli/health-check.js || exit 1

ENTRYPOINT ["node", "/app/dist/bin/duckbrain.js"]
CMD ["http", "--port=3000", "--bind=0.0.0.0"]
```

docker-compose.yml (single container):
```yaml
version: '3.8'

services:
  duckbrain:
    build: .
    ports:
      - "127.0.0.1:3000:3000"  # Bind to localhost only (D-20)
    volumes:
      - duckbrain-data:/data    # Named volume (D-07 preferred)
    environment:
      - NODE_ENV=production
      - DUCKBRAIN_DATA_DIR=/data
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "node", "/app/dist/cli/health-check.js"]
      interval: 30s
      timeout: 3s
      retries: 3

volumes:
  duckbrain-data:
```

docker-compose.multi.yml (multi-container with DuckDB separation):
```yaml
version: '3.8'

services:
  duckbrain:
    build: .
    ports:
      - "127.0.0.1:3000:3000"
    volumes:
      - duckbrain-data:/data
    environment:
      - NODE_ENV=production
      - DUCKBRAIN_DATA_DIR=/data
      - DUCKDB_HOST=duckdb
      - DUCKDB_PORT=5432
    depends_on:
      - duckdb
    restart: unless-stopped

  duckdb:
    image: duckdb/duckdb:latest
    volumes:
      - duckbrain-data:/data
    environment:
      - DUCKDB_DATABASE=/data/duckbrain.db
    # DuckDB runs in-process, so this is illustrative
    # In practice, DuckDB is embedded in the Node.js app
    command: ["tail", "-f", "/dev/null"]  # Keep container alive

volumes:
  duckbrain-data:
```

Note: DuckDB is embedded, so multi-container is primarily for illustration. Actual multi-container benefits:
- Separate volumes for different data types
- Resource isolation between app and database
- Independent scaling (if DuckDB becomes server-based later)
