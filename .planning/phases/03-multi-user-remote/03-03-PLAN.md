```typescript
interface RateLimitStore {
  [ip: string]: {
    tokens: number;
    lastRefill: number;
  };
}

const store: RateLimitStore = {};

export function rateLimitMiddleware(config: RateLimitConfig) {
  return (req: Request, res: Response, next: NextFunction) => {
    const ip = req.ip || 'unknown';
    const now = Date.now();
    const windowMs = 60000; // 1 minute
    
    if (!store[ip]) {
      store[ip] = { tokens: config.requestsPerMinute, lastRefill: now };
    }
    
    // Refill tokens
    const elapsed = now - store[ip].lastRefill;
    const tokensToAdd = Math.floor(elapsed / (windowMs / config.requestsPerMinute));
    store[ip].tokens = Math.min(
      config.requestsPerMinute,
      store[ip].tokens + tokensToAdd
    );
    
    if (store[ip].tokens > 0) {
      store[ip].tokens--;
      next();
    } else {
      res.status(429).json({ error: 'Rate limit exceeded' });
    }
  };
}
```

import { execSync } from 'child_process';
import fs from 'fs';
import path from 'path';
import os from 'os';

export async function installService(options: { system?: boolean }): Promise<boolean> {
  const isSystem = options.system || false;
  
  if (isSystem && process.getuid?.() !== 0) {
    console.log('System-wide service requires sudo:');
    console.log('  sudo duckbrain service install --system');
    return false;
  }
  
  const serviceDir = isSystem 
    ? '/etc/systemd/system'
    : path.join(os.homedir(), '.config/systemd/user');
  
  // Create user service directory if needed
  if (!isSystem) {
    fs.mkdirSync(serviceDir, { recursive: true });
  }
  
  const servicePath = path.join(serviceDir, 'duckbrain.service');
  const execPath = process.argv[1]; // Current executable path
  
  const serviceContent = `[Unit]
Description=DuckBrain MCP Server
After=network.target

[Service]
Type=simple
User=${isSystem ? 'duckbrain' : '%I'}
ExecStart=${execPath} http --port=3000
Restart=always
RestartSec=10

[Install]
WantedBy=${isSystem ? 'multi-user.target' : 'default.target'}
`;
  
  fs.writeFileSync(servicePath, serviceContent);
  
  console.log(`Service installed to ${servicePath}`);
  console.log('Run: duckbrain service start');
  return true;
}
